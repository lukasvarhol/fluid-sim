#version 330 core

in vec4 fragmentColor;

out vec4 screenColor;

in vec2 uv;

uniform float aspect;
uniform vec4 color;

void main() {
    vec2 p = uv;

    if (aspect >= 1.0) {
        p.x *= aspect;
    } else {
        p.y /= aspect;  
    }

    if (dot(p, p) > 1.0) discard;

    screenColor = color;
}
